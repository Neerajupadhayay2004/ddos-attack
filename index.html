⚡Locust DDoS Simulation
# Step 1: Install Locust
$ pip install locust

# Step 2: Run with 500 users, 50/sec spawn rate
$ locust -f locustfile.py --headless -u 500 -r 50 --host=http://localhost:5000

# Step 3: Or use Locust Web UI
$ locust -f locustfile.py --host=http://localhost:5000
# Then open http://localhost:8089
# Manual curl flood (bash) — 1000 requests fast
$ for i in $(seq 1 1000); do curl -s http://localhost:5000/api/test &; done

# Python requests flood
$ python3 -c "import requests,threading; [threading.Thread(target=lambda:requests.get('http://localhost:5000/api/test')).start() for _ in range(500)]"

# Apache Bench (ab) — 5000 requests, 100 concurrent
$ ab -n 5000 -c 100 http://localhost:5000/api/test
